134 void T6Model::setup(tgWorld& world)
135 {
136 
137  const tgRod::Config rodConfig(c.radius, c.density);
138  tgLinearString::Config muscleConfig(c.stiffness, c.damping);
139 
140  tgStructure s;
141  addNodes(s);
142  addRods(s);
143  addMuscles(s);
144  s.move(btVector3(0, 10, 0));
145 
146  // Create the build spec that uses tags to turn the structure into a real model
147  tgBuildSpec spec;
148  spec.addBuilder("rod", new tgRodInfo(rodConfig));
149  spec.addBuilder("muscle", new tgLinearStringInfo(muscleConfig));
150 
151  // Create your structureInfo
152  tgStructureInfo structureInfo(s, spec);
153 
154  // Use the structureInfo to build ourselves
155  structureInfo.buildInto(*this, world);
156 
157  // We could now use tgCast::filter or similar to pull out the
158  // models (e.g. muscles) that we want to control.
159  allMuscles = tgCast::filter<tgModel, tgLinearString> (getDescendants());
160 
161  // Actually setup the children
162  tgModel::setup(world);
163 }

165 void T6Model::step(double dt)
166 {
167  // Precondition
168  if (dt <= 0.0)
169  {
170  throw std::invalid_argument("dt is not positive");
171  }
172  else
173  {
174  // Notify observers (controllers) of the step so that they can take action
175  notifyStep(dt);
176  tgModel::step(dt); // Step any children
177  }
178 }
